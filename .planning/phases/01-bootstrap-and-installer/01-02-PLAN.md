  <files>src/installer/copy-files.js, src/installer/version.js</files>
  <action>
Create src/installer/copy-files.js:
- Import fs-extra and path
- Import { getInstallDir, getSkillsDir } from '../utils/paths'
- Export async function copySkills(sourceDir):
  - sourceDir defaults to path.join(__dirname, '../../skills')
  - Reads all entries in sourceDir
  - For each directory entry, copies to path.join(getSkillsDir(), entry.name)
  - Uses fs.copy with { overwrite: false, errorOnExist: false } (install should not overwrite)
  - Returns array of copied skill directory names
- Export async function copyTemplates(sourceDir):
  - sourceDir defaults to path.join(__dirname, '../../templates')
  - Copies to path.join(getInstallDir(), 'templates')
  - Uses fs.copy with { overwrite: false }
- Export async function copyReferences(sourceDir):
  - sourceDir defaults to path.join(__dirname, '../../references')
  - Copies to path.join(getInstallDir(), 'references')
  - Uses fs.copy with { overwrite: false }

Create src/installer/version.js:
- Import fs-extra, path
- Import { getInstallDir } from '../utils/paths'
- Import { SCHEMA_VERSION, writeJsonWithSchema } from '../utils/schema'
- Export async function writeVersion(version):
  - version param is the dwaVersion string (from package.json)
  - Writes to path.join(getInstallDir(), '.dwa-version')
  - Uses writeJsonWithSchema with data: { dwaVersion: version, installedAt: new Date().toISOString() }
  - This ensures schemaVersion is automatically included (via writeJsonWithSchema)
- Export async function readVersion():
  - Reads and parses path.join(getInstallDir(), '.dwa-version')
  - If file doesn't exist, returns { schemaVersion: '1.0.0', dwaVersion: '0.0.0' } (legacy fallback)
  - Returns parsed JSON object
  </action>

Create src/commands/install.js:
- Import fs-extra, path, ora (v5.x CJS-compatible)
- Import { getInstallDir, getSkillsDir } from '../utils/paths'
- Import { copySkills, copyTemplates, copyReferences } from '../installer/copy-files'
- Import { writeVersion } from '../installer/version'
- Read package.json version: require('../../package.json').version

Export async function install():
1. Create spinner: ora('Installing DWA...').start()
2. Check if already installed:
   - if (await fs.pathExists(getInstallDir())): spinner.fail('DWA already installed. Use --upgrade to update.'); process.exit(1)
3. Create install directory: await fs.ensureDir(getInstallDir())
4. spinner.text = 'Copying skills...'
5. await copySkills() -- copy skills to ~/.claude/skills/dwa-*/
6. spinner.text = 'Copying templates...'
7. await copyTemplates() -- copy templates to ~/.claude/dwa/templates/
8. spinner.text = 'Copying references...'
9. await copyReferences() -- copy references to ~/.claude/dwa/references/
10. spinner.text = 'Writing version info...'
11. await writeVersion(packageVersion) -- write .dwa-version
12. spinner.succeed('DWA installed successfully!')
13. Print summary:
    - console.log(`\n  Install directory: ${getInstallDir()}`)
    - console.log(`  Skills directory:  ${getSkillsDir()}`)
    - console.log('\n  Run /dwa:* commands in Claude Code to get started.\n')
14. Wrap entire body in try/catch:
    - catch: spinner.fail('Installation failed: ' + error.message); process.exit(1)
  </action>
