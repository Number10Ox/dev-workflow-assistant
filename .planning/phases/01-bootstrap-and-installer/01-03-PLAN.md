  <files>src/installer/backup.js, src/commands/upgrade.js</files>
  <action>
Create src/installer/backup.js:
- Import fs-extra
- Export async function backupInstallation(installDir):
  - Generate timestamp: new Date().toISOString().replace(/[:.]/g, '-')
  - backupDir = `${installDir}.backup.${timestamp}`
  - await fs.copy(installDir, backupDir, { errorOnExist: true, preserveTimestamps: true })
  - Return backupDir path (caller may want to display it)

Create src/commands/upgrade.js:
- Import fs-extra, ora, semver
- Import { getInstallDir } from '../utils/paths'
- Import { copySkills, copyTemplates, copyReferences } from '../installer/copy-files'
- Import { readVersion, writeVersion } from '../installer/version'
- Import { backupInstallation } from '../installer/backup'
- Read package version from ../../package.json

Export async function upgrade():
1. const installDir = getInstallDir()
2. const spinner = ora('Checking installation...').start()
3. Check not installed:
   - if (!await fs.pathExists(installDir)): spinner.fail('DWA not installed. Run with --install first.'); process.exit(1)
4. Read current version: const current = await readVersion()
5. Compare versions using semver:
   - const packageVersion = require('../../package.json').version
   - if (!semver.lt(current.dwaVersion, packageVersion)):
     spinner.info(`Already on latest version (${current.dwaVersion})`); return
6. spinner.text = `Upgrading from ${current.dwaVersion} to ${packageVersion}...`
7. Create backup:
   - spinner.text = 'Creating backup...'
   - const backupDir = await backupInstallation(installDir)
8. Copy files with OVERWRITE (unlike install):
   - spinner.text = 'Updating skills...'
   - Call copySkills() -- but need overwrite version
   - spinner.text = 'Updating templates...'
   - Call copyTemplates() with overwrite option
   - spinner.text = 'Updating references...'
   - Call copyReferences() with overwrite option

   NOTE: copy-files.js functions use { overwrite: false } for install. For upgrade, add an optional `overwrite` parameter (default false) to each function. Upgrade calls with overwrite=true.

   UPDATE src/installer/copy-files.js: Add `options = {}` param to each function. Use `{ overwrite: options.overwrite || false }` in fs.copy calls. This is a backward-compatible change.

9. Write new version: await writeVersion(packageVersion)
10. spinner.succeed(`Upgraded from ${current.dwaVersion} to ${packageVersion}`)
11. console.log(`\n  Backup saved to: ${backupDir}\n`)
12. Wrap in try/catch: spinner.fail('Upgrade failed: ' + error.message); console.log('Restore from backup if needed.'); process.exit(1)
  </action>

<task type="auto">
  <name>Task 2: Implement uninstall command</name>
  <files>src/commands/uninstall.js</files>
  <action>
Create src/commands/uninstall.js:
- Import fs-extra, path, ora
- Import { getInstallDir, getSkillsDir } from '../utils/paths'

Export async function uninstall():
1. const installDir = getInstallDir()
2. const skillsDir = getSkillsDir()
3. const spinner = ora('Uninstalling DWA...').start()
4. try block:
   a. Track what was removed for summary
   b. Remove install directory:
      - if (await fs.pathExists(installDir)):
        await fs.remove(installDir)
        removedInstall = true
   c. Remove skill directories (only dwa-* prefixed):
      - if (await fs.pathExists(skillsDir)):
        const entries = await fs.readdir(skillsDir)
        const dwaSkills = entries.filter(e => e.startsWith('dwa-'))
        for (const dir of dwaSkills):
          await fs.remove(path.join(skillsDir, dir))
        removedSkills = dwaSkills.length
   d. If nothing was found:
      - spinner.info('DWA was not installed. Nothing to remove.')
      - return
   e. spinner.succeed('DWA uninstalled successfully')
   f. Print summary:
      - if removedInstall: console.log(`  Removed: ${installDir}`)
      - if removedSkills > 0: console.log(`  Removed: ${removedSkills} skill(s) from ${skillsDir}`)
      - console.log('')
5. catch block:
   - spinner.fail('Uninstall failed: ' + error.message)
   - process.exit(1)
  </action>
  <verify>
- `node -e "const {uninstall} = require('./src/commands/uninstall'); console.log(typeof uninstall)"` outputs "function"
  </verify>
  <done>Uninstall command removes install directory and all dwa-* skill directories; handles not-installed case gracefully</done>
</task>
