   ```typescript
   import { Version3Client } from 'jira.js';
   import { backOff } from 'exponential-backoff';

   // Re-export interface types for consumers
   export interface Issue {
     id: string;
     title: string;
     description: string;
     status: string;
     assignee?: string;
     labels?: string[];
     url?: string;
     externalId?: string;
     identifier?: string;
     container?: string;
   }

   export interface IssueCreateInput {
     title: string;
     description: string;
     labels?: string[];
     externalId?: string;
     container?: string;
   }

   export interface IssueFilter {
     externalId?: string;
     container?: string;
     status?: string;
   }

   export interface IssueTracker {
     createIssue(input: IssueCreateInput): Promise<Issue>;
     updateIssue(id: string, updates: Partial<Issue>): Promise<Issue>;
     getIssue(id: string): Promise<Issue>;
     listIssues(filter?: IssueFilter): Promise<Issue[]>;
     queryByExternalId(externalId: string): Promise<Issue | null>;
   }

   export class JiraTracker implements IssueTracker {
     private client: Version3Client;
     private defaultProject: string;
     private host: string;

     constructor(host: string, email: string, apiToken: string, defaultProject: string) {
       this.host = host.replace(/^https?:\/\//, '').replace(/\/$/, '');
       this.defaultProject = defaultProject;

       this.client = new Version3Client({
         host: `https://${this.host}`,
         authentication: {
           basic: {
             email,
             apiToken,
           },
         },
       });
     }

     private async withRateLimitHandling<T>(operation: () => Promise<T>): Promise<T> {
       return backOff(
         async () => {
           try {
             return await operation();
           } catch (err: any) {
             // JIRA returns 429 for rate limiting
             if (err.status === 429) {
               throw err; // Trigger retry
             }
             // Don't retry client errors
             if (err.status >= 400 && err.status < 500) {
               throw err;
             }
             throw err;
           }
         },
         {
           numOfAttempts: 5,
           startingDelay: 1000,
           timeMultiple: 2,
           jitter: 'full',
           maxDelay: 30000,
           retry: (err: any) => {
             return err.status === 429 || err.status >= 500;
           },
         }
       );
     }

     async createIssue(input: IssueCreateInput): Promise<Issue> {
       return this.withRateLimitHandling(async () => {
         const projectKey = input.container || this.defaultProject;

         const issue = await this.client.issues.createIssue({
           fields: {
             project: { key: projectKey },
             summary: input.title,
             description: {
               type: 'doc',
               version: 1,
               content: [
                 {
                   type: 'paragraph',
                   content: [{ type: 'text', text: input.description || '' }],
                 },
               ],
             },
             issuetype: { name: 'Task' },
             labels: input.labels,
           },
         });

         // Store external ID in issue properties for deduplication
         if (input.externalId && issue.id) {
           await this.setExternalId(issue.id, input.externalId);
         }

         return this.mapJiraIssueToIssue(issue, input.externalId);
       });
     }

     async updateIssue(id: string, updates: Partial<Issue>): Promise<Issue> {
       return this.withRateLimitHandling(async () => {
         const fields: any = {};

         if (updates.title) {
           fields.summary = updates.title;
         }

         if (updates.description) {
           fields.description = {
             type: 'doc',
             version: 1,
             content: [
               {
                 type: 'paragraph',
                 content: [{ type: 'text', text: updates.description }],
               },
             ],
           };
         }

         if (updates.labels) {
           fields.labels = updates.labels;
         }

         await this.client.issues.editIssue({
           issueIdOrKey: id,
           fields,
         });

         return this.getIssue(id);
       });
     }

     async getIssue(id: string): Promise<Issue> {
       return this.withRateLimitHandling(async () => {
         const issue = await this.client.issues.getIssue({
           issueIdOrKey: id,
           fields: ['summary', 'description', 'status', 'assignee', 'labels', 'project'],
         });

         const externalId = await this.getExternalId(issue.id!);
         return this.mapJiraIssueToIssue(issue, externalId);
       });
     }

     async listIssues(filter?: IssueFilter): Promise<Issue[]> {
       return this.withRateLimitHandling(async () => {
         const jqlParts: string[] = [];

         if (filter?.container) {
           jqlParts.push(`project = "${filter.container}"`);
         } else if (this.defaultProject) {
           jqlParts.push(`project = "${this.defaultProject}"`);
         }

         if (filter?.status) {
           jqlParts.push(`status = "${filter.status}"`);
         }

         const jql = jqlParts.length > 0 ? jqlParts.join(' AND ') : '';

         const result = await this.client.issueSearch.searchForIssuesUsingJql({
           jql,
           fields: ['summary', 'description', 'status', 'assignee', 'labels', 'project'],
         });

         const issues: Issue[] = [];
         for (const issue of result.issues || []) {
           const externalId = await this.getExternalId(issue.id!);
           issues.push(this.mapJiraIssueToIssue(issue, externalId));
         }

         return issues;
       });
     }

     async queryByExternalId(externalId: string): Promise<Issue | null> {
       return this.withRateLimitHandling(async () => {
         // Search using entity properties
         // JIRA doesn't support JQL on properties, so we search by label fallback
         const jql = `labels = "dwa:${externalId}" OR labels = "dwa-external-id:${externalId}"`;

         const result = await this.client.issueSearch.searchForIssuesUsingJql({
           jql,
           fields: ['summary', 'description', 'status', 'assignee', 'labels', 'project'],
           maxResults: 1,
         });

         if (!result.issues || result.issues.length === 0) {
           return null;
         }

         const issue = result.issues[0];
         return this.mapJiraIssueToIssue(issue, externalId);
       });
     }

     /**
      * Store external ID in issue properties.
      */
     private async setExternalId(issueId: string, externalId: string): Promise<void> {
       try {
         await this.client.issueProperties.setIssueProperty({
           issueIdOrKey: issueId,
           propertyKey: 'dwa-external-id',
           propertyValue: externalId,
         } as any);

         // Also add as label for JQL searchability
         const issue = await this.client.issues.getIssue({
           issueIdOrKey: issueId,
           fields: ['labels'],
         });

         const existingLabels = issue.fields?.labels || [];
         await this.client.issues.editIssue({
           issueIdOrKey: issueId,
           fields: {
             labels: [...existingLabels, `dwa:${externalId}`],
           },
         });
       } catch (err) {
         console.error('Failed to set external ID:', err);
         // Non-fatal - issue was created, just without external ID
       }
     }

     /**
      * Get external ID from issue properties.
      */
     private async getExternalId(issueId: string): Promise<string | undefined> {
       try {
         const prop = await this.client.issueProperties.getIssueProperty({
           issueIdOrKey: issueId,
           propertyKey: 'dwa-external-id',
         });
         return prop.value as string;
       } catch {
         return undefined;
       }
     }

     private mapJiraIssueToIssue(jiraIssue: any, externalId?: string): Issue {
       const fields = jiraIssue.fields || {};

       // Extract description text from ADF format
       let description = '';
       if (fields.description?.content) {
         description = fields.description.content
           .map((block: any) =>
             block.content?.map((item: any) => item.text || '').join('') || ''
           )
           .join('\n');
       }

       return {
         id: jiraIssue.id,
         title: fields.summary || '',
         description,
         status: fields.status?.name || 'Unknown',
         assignee: fields.assignee?.accountId,
         labels: fields.labels || [],
         url: `https://${this.host}/browse/${jiraIssue.key}`,
         externalId,
         identifier: jiraIssue.key,
         container: fields.project?.key,
       };
     }
   }
   ```

   import * as vscode from 'vscode';

   export class JiraAuth {
     private secrets: vscode.SecretStorage;

     constructor(secrets: vscode.SecretStorage) {
       this.secrets = secrets;
     }

     async getApiToken(): Promise<string | undefined> {
       return this.secrets.get('jira-api-token');
     }

     async setApiToken(token: string): Promise<void> {
       await this.secrets.store('jira-api-token', token);
     }

     async clearApiToken(): Promise<void> {
       await this.secrets.delete('jira-api-token');
     }

     getConfiguration() {
       const config = vscode.workspace.getConfiguration('jiraProvider');
       return {
         host: config.get<string>('host') || '',
         email: config.get<string>('email') || '',
         defaultProject: config.get<string>('defaultProject') || '',
       };
     }

     async isConfigured(): Promise<boolean> {
       const config = this.getConfiguration();
       const token = await this.getApiToken();
       return !!(config.host && config.email && token);
     }
   }

2. Create `packages/jira-provider/src/extension.ts`:

   ```typescript
   import * as vscode from 'vscode';
   import { JiraTracker, IssueTracker, Issue, IssueCreateInput, IssueFilter } from './jiraTracker';
   import { JiraAuth } from './auth';

   let tracker: JiraTracker | null = null;
   let auth: JiraAuth;

   export interface JiraProviderAPI extends IssueTracker {
     checkAvailability(): Promise<{ available: boolean; error?: string; setupInstructions?: string }>;
   }

   export async function activate(context: vscode.ExtensionContext): Promise<JiraProviderAPI> {
     auth = new JiraAuth(context.secrets);

     // Register configure command
     context.subscriptions.push(
       vscode.commands.registerCommand('jiraProvider.configure', async () => {
         const token = await vscode.window.showInputBox({
           prompt: 'Enter your JIRA API token',
           password: true,
           placeHolder: 'API token from id.atlassian.com',
         });

         if (token) {
           await auth.setApiToken(token);
           vscode.window.showInformationMessage('JIRA API token saved.');
           await initializeTracker();
         }
       })
     );

     // Try to initialize on activation
     await initializeTracker();

     // Return API
     return {
       createIssue: (input: IssueCreateInput) => ensureTracker().createIssue(input),
       updateIssue: (id: string, updates: Partial<Issue>) => ensureTracker().updateIssue(id, updates),
       getIssue: (id: string) => ensureTracker().getIssue(id),
       listIssues: (filter?: IssueFilter) => ensureTracker().listIssues(filter),
       queryByExternalId: (externalId: string) => ensureTracker().queryByExternalId(externalId),
       checkAvailability,
     };
   }

   async function initializeTracker(): Promise<void> {
     const config = auth.getConfiguration();
     const token = await auth.getApiToken();

     if (config.host && config.email && token) {
       tracker = new JiraTracker(config.host, config.email, token, config.defaultProject);
       console.log('JIRA Provider: Tracker initialized');
     } else {
       tracker = null;
       console.log('JIRA Provider: Not configured');
     }
   }

   function ensureTracker(): JiraTracker {
     if (!tracker) {
       throw new Error('JIRA not configured. Run "JIRA Provider: Configure" command.');
     }
     return tracker;
   }

   async function checkAvailability(): Promise<{
     available: boolean;
     error?: string;
     setupInstructions?: string;
   }> {
     const isConfigured = await auth.isConfigured();

     if (!isConfigured) {
       return {
         available: false,
         error: 'JIRA not configured',
         setupInstructions: `
To configure JIRA integration:

1. Get an API token from https://id.atlassian.com/manage-profile/security/api-tokens
2. Open VS Code Settings and set:
   - jiraProvider.host: your-domain.atlassian.net
   - jiraProvider.email: your-email@example.com
   - jiraProvider.defaultProject: PROJ (your project key)
3. Run command "JIRA Provider: Configure" and paste your API token
`.trim(),
       };
     }

     return { available: true };
   }

   export function deactivate(): void {
     tracker = null;
   }
   ```

   ```markdown
   # Extending DWA: Adding Issue Tracker Providers

   DWA uses a provider-agnostic architecture for issue tracker integration.
   This document explains how to add support for new issue trackers.

   ## Architecture Overview

   ```
   DWA Core
      │
      ▼
   Bridge Client (src/linear/bridge-client.js)
      │
      ▼ discovers via VS Code extension API
   Provider Extension (e.g., linear-provider, jira-provider)
      │
      ▼ implements IssueTracker interface
   Issue Tracker API (Linear, JIRA, GitHub, etc.)
   ```

   ## The IssueTracker Interface

   Any provider must implement this interface:

   ```typescript
   interface IssueTracker {
     createIssue(input: IssueCreateInput): Promise<Issue>;
     updateIssue(id: string, updates: Partial<Issue>): Promise<Issue>;
     getIssue(id: string): Promise<Issue>;
     listIssues(filter?: IssueFilter): Promise<Issue[]>;
     queryByExternalId(externalId: string): Promise<Issue | null>;
   }
   ```

   ## Adding a New Provider

   ### Step 1: Create VS Code Extension

   Create a new package in `devex-service-bridge/packages/`:

   ```
   packages/
     your-provider/
       package.json
       src/
         extension.ts    # VS Code entry point
         yourTracker.ts  # IssueTracker implementation
         auth.ts         # Credential management
   ```

   ### Step 2: Implement IssueTracker

   See `linearTracker.ts` or `jiraTracker.ts` for reference implementations.

   Key considerations:
   - Use `exponential-backoff` for rate limit handling
   - Store credentials in `vscode.SecretStorage`
   - Map your API's concepts to the generic interface
   - Implement `externalId` support for deduplication

   ### Step 3: Export the API

   Your extension's `activate()` must return an object implementing IssueTracker:

   ```typescript
   export async function activate(context: ExtensionContext): Promise<IssueTracker> {
     // Initialize and return your tracker
   }
   ```

   ### Step 4: Register with DWA

   Add your extension ID to `ISSUE_TRACKER_PROVIDER_IDS` in
   `dev-workflow-assistant/src/linear/bridge-client.js`:

   ```javascript
   const ISSUE_TRACKER_PROVIDER_IDS = [
     'jedwards.your-provider',
     'your-provider',  // Development mode
   ];
   ```

   ## Testing

   DWA will automatically discover your provider when:
   1. The extension is installed in VS Code
   2. The extension is activated
   3. The exported API has the required methods

   Test with:
   ```bash
   dwa --sync-linear --dry-run
   ```

   ## Examples

   - **Linear**: `packages/linear-provider/` - Reference implementation
   - **JIRA**: `packages/jira-provider/` - Cloud JIRA with entity properties
   ```
