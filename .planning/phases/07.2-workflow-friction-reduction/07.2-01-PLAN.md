### Task 2: CLI Commands

Wire maintenance utilities into CLI entry point:

```javascript
// In bin/dwa.js
case 'clean':
  const maint = new DWAMaintenance();
  if (args.includes('--all')) {
    await maint.removeAllState();
  } else {
    const result = await maint.cleanupOrphaned();
    console.log(`Cleaned ${result.cleaned} orphaned deliverables`);
  }
  break;

case 'validate':
  const validation = await maint.validateState();
  if (validation.valid) {
    console.log('DWA state is valid');
  } else {
    console.log('Issues found:');
    validation.issues.forEach(i => console.log(`  - ${i}`));
    process.exit(1);
  }
  break;

case 'stats':
  const stats = await maint.getStats();
  console.log(`Deliverables: ${stats.deliverables} (${stats.orphaned} orphaned)`);
  console.log(`Packets: ${stats.packets}`);
  console.log(`Disk usage: ${stats.diskUsage}`);
  break;
```

**Files**: `bin/dwa.js`

**Tests**:
- `dwa clean` invokes cleanupOrphaned
- `dwa clean --all` removes all state
- `dwa validate` exits 0 on valid, 1 on invalid
- `dwa stats` displays formatted statistics

### Task 3: Staleness Detection in Packet Generation

Add freshness check before packet generation:

```javascript
// In src/packet-generator.js or execution packet command
async function generatePacket(deliverableId, options = {}) {
  const maint = new DWAMaintenance();
  const freshness = await maint.checkFreshness();

  if (freshness.stale) {
    if (options.force) {
      console.log('Warning: Registry may be stale. Using --force.');
    } else {
      console.log('Spec has been modified since last parse.');
      console.log('Run `dwa parse` to update, or use --force to continue.');
      return { success: false, reason: 'stale' };
    }
  }

}
```

### Task 4: Gitignore Integration

Update install command to handle `.gitignore`:

```javascript
// In install.js
async function ensureGitignore(projectRoot) {
  const gitignorePath = path.join(projectRoot, '.gitignore');
  const dwaEntry = '.dwa/';

  if (await fs.pathExists(gitignorePath)) {
    const content = await fs.readFile(gitignorePath, 'utf8');
    if (!content.includes(dwaEntry)) {
      // Append with comment
      await fs.appendFile(gitignorePath, `\n# DWA state (regenerated from source files)\n${dwaEntry}\n`);
      console.log('Added .dwa/ to .gitignore');
    }
  } else {
    // Create minimal .gitignore
    await fs.writeFile(gitignorePath, `# DWA state (regenerated from source files)\n${dwaEntry}\n`);
    console.log('Created .gitignore with .dwa/ entry');
  }
}
```

**Files**: `src/install.js` (or equivalent)

**Tests**:
- Install appends to existing .gitignore without .dwa/
- Install creates .gitignore if missing
- Install preserves existing .gitignore content
- Install is idempotent (doesn't duplicate entry)
