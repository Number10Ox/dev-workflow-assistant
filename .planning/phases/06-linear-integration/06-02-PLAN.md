   private async withRateLimitHandling<T>(operation: () => Promise<T>): Promise<T> {
     return backOff(
       async () => {
         try {
           return await operation();
         } catch (err) {
           if (err instanceof LinearError) {
             const rateLimitError = err.errors?.find(
               (e: any) => e.extensions?.code === 'RATELIMITED'
             );
             if (rateLimitError) {
               throw err; // Trigger backoff retry
             }
             if (err instanceof InvalidInputLinearError) {
               throw err; // Don't retry input errors
             }
           }
           throw err;
         }
       },
       {
         numOfAttempts: 5,
         startingDelay: 1000,
         timeMultiple: 2,
         jitter: 'full',
         maxDelay: 30000,
         retry: (err: any) => {
           if (err instanceof InvalidInputLinearError) return false;
           if (err instanceof LinearError) {
             return err.errors?.some((e: any) => e.extensions?.code === 'RATELIMITED') ?? false;
           }
           return true;
         },
       }
     );
   }
   ```

   async createIssue(input: IssueCreateInput): Promise<Issue> {
     return this.withRateLimitHandling(async () => {
       const teams = await this.client.teams();
       const team = teams.nodes[0];
       if (!team) {
         throw new Error('No teams found in Linear workspace');
       }

       const issueCreate = await this.client.createIssue({
         teamId: team.id,
         title: input.title,
         description: input.description,
         externalId: input.externalId,
         projectId: input.container,  // Map container to projectId
       });

       if (!issueCreate.success || !issueCreate.issue) {
         throw new Error('Failed to create Linear issue');
       }

       return this.mapLinearIssueToIssue(issueCreate.issue);
     });
   }
   ```

5. Add queryByExternalId method:

   async queryByExternalId(externalId: string): Promise<Issue | null> {
     return this.withRateLimitHandling(async () => {
       const issues = await this.client.issues({
         filter: { externalId: { eq: externalId } },
       });

       if (issues.nodes.length === 0) {
         return null;
       }

       return this.mapLinearIssueToIssue(issues.nodes[0]);
     });
   }
   ```

   async listIssues(filter?: IssueFilter): Promise<Issue[]> {
     return this.withRateLimitHandling(async () => {
       const linearFilter: any = {};

       if (filter?.status) {
         linearFilter.state = { name: { eq: filter.status } };
       }
       if (filter?.externalId) {
         linearFilter.externalId = { eq: filter.externalId };
       }
       if (filter?.container) {
         linearFilter.project = { id: { eq: filter.container } };
       }

       const issues = await this.client.issues({
         filter: Object.keys(linearFilter).length > 0 ? linearFilter : undefined,
       });

       return issues.nodes.map((issue) => this.mapLinearIssueToIssue(issue));
     });
   }
   ```
