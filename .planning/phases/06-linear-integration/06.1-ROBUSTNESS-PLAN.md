     ```javascript
     /**
      * Fallback dedupe: search for DWA marker in issue descriptions.
      * Used when externalId query fails or is unsupported.
      *
      * @param {Object} client - Bridge client
      * @param {string} externalId - The externalId to search for
      * @param {string} [container] - Optional container/project to narrow search
      * @returns {Promise<Issue|null>}
      */
     async function findIssueByDwaMarker(client, externalId, container) {
       // Search issues in container
       const issues = await client.listIssues({
         container,
         status: null  // All statuses
       });

       // Look for DWA marker containing this externalId
       const marker = `<!-- DWA:SYNC externalId="${externalId}"`;
       for (const issue of issues) {
         if (issue.description?.includes(marker)) {
           return issue;
         }
       }

       return null;
     }
     ```

   - Update `findExistingIssue()` with fallback chain:
     ```javascript
     async function findExistingIssue(client, deliverable, container) {
       // 1. Fast path: registry has linear_issue_id
       if (deliverable.linear_issue_id) {
         try {
           const issue = await client.getIssue(deliverable.linear_issue_id);
           if (issue) return { issue, source: 'registry' };
         } catch (e) {
           console.log(`Issue ${deliverable.linear_issue_id} not found, trying externalId`);
         }
       }

       // 2. Primary lookup: externalId query (if supported)
       const externalId = generateExternalId(deliverable);
       const supportsQuery = await client.supportsFeature('supportsExternalIdQuery');

       if (supportsQuery) {
         try {
           const issue = await client.queryByExternalId(externalId);
           if (issue) return { issue, source: 'externalId' };
         } catch (e) {
           console.warn(`externalId query failed: ${e.message}, trying marker fallback`);
         }
       }

       // 3. Fallback: search for DWA marker in issue body
       const issue = await findIssueByDwaMarker(client, externalId, container);
       if (issue) return { issue, source: 'marker-search' };

       return { issue: null, source: null };
     }
     ```

   - Add `computeDiffSnippet(oldContent, newContent, contextLines)`:
     ```javascript
     /**
      * Compute a short diff snippet for conflict reporting.
      * Shows first differing region with context lines.
      *
      * @param {string} oldContent
      * @param {string} newContent
      * @param {number} contextLines - Lines of context around diff
      * @returns {string} Diff snippet (truncated to ~10 lines)
      */
     function computeDiffSnippet(oldContent, newContent, contextLines = 3) {
       const oldLines = oldContent.split('\n');
       const newLines = newContent.split('\n');

       // Find first differing line
       let diffStart = 0;
       while (diffStart < oldLines.length && diffStart < newLines.length) {
         if (oldLines[diffStart] !== newLines[diffStart]) break;
         diffStart++;
       }

       if (diffStart >= oldLines.length && diffStart >= newLines.length) {
         return '(no differences found)';
       }

       // Extract snippet with context
       const start = Math.max(0, diffStart - contextLines);
       const end = Math.min(diffStart + contextLines + 1, Math.max(oldLines.length, newLines.length));

       let snippet = `@@ line ${start + 1} @@\n`;
       for (let i = start; i < end; i++) {
         const oldLine = oldLines[i] ?? '';
         const newLine = newLines[i] ?? '';
         if (oldLine === newLine) {
           snippet += ` ${oldLine}\n`;
         } else {
           if (oldLines[i] !== undefined) snippet += `-${oldLine}\n`;
           if (newLines[i] !== undefined) snippet += `+${newLine}\n`;
         }
       }

       return snippet.slice(0, 500);  // Truncate long diffs
     }
     ```
