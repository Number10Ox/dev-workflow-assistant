```typescript
/**
 * Bridge capability declaration for consumer compatibility checks.
 * Consumers (like DWA) check capabilities before invoking methods.
 */

export interface BridgeCapabilities {
  /** Semantic version of the bridge API */
  apiVersion: string;

  /** Available issue tracker providers (e.g., ['linear', 'jira']) */
  providers: string[];

  /** Feature flags indicating supported capabilities */
  features: {
    externalId: boolean;         // Supports externalId for deduplication
    container: boolean;          // Supports container/project grouping
    queryByExternalId: boolean;  // Supports lookup by externalId
  };
}

/**
 * Interface for providers that can report capabilities.
 */
export interface CapabilityProvider {
  getCapabilities(): BridgeCapabilities;
}

/**
 * Current bridge API version.
 * Increment when breaking changes are made to IssueTracker interface.
 */
export const BRIDGE_API_VERSION = '2.0.0';

/**
 * Default capabilities for this version of the bridge.
 */
export function getDefaultCapabilities(providers: string[]): BridgeCapabilities {
  return {
    apiVersion: BRIDGE_API_VERSION,
    providers,
    features: {
      externalId: true,
      container: true,
      queryByExternalId: true,
    },
  };
}
```
